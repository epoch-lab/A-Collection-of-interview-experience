1，forEach中实现setTimeout延迟输出
```js
const foo = (n) => {
  return new Promise((resolve, reject) => {
    setTimeout(() => {
      resolve(n);
    }, 1000);
  });
};

const test = (nums) => {
  nums.forEach(async (n) => {
    let num = await foo(n);
    console.log(num);
  });
};

test([1, 2, 3]);

// 输出结果
// 1
// 2
// 3

```
